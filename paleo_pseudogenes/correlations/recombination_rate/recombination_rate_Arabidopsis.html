Required libraries

library(MareyMap)
library(corrr)

Background

Here, we estimate the recombination rate along the genome (in bins) using the Marey Map approach. Next, we then assess whether there is a (negative) correlation between recombination rate and pseudogene number.

paper with Arabidopsis thaliana dataset: https://doi.org/10.3389/fgene.2017.00201

Reference of recombination landscape inference in flowering plants: https://doi.org/10.1371/journal.pgen.1010141

Aim

Find relationship between pseudogenes and recombination rate

Data

# load in data

# recombination rate data
df <- readxl::read_excel("./ath/arabidopsis_markers.xlsx", sheet = 2) %>%
  mutate(set = "Arabidopsis thaliana") %>%
  # convert to input format of the MareyMap package
  rename(map = chromosome, mkr = bin_markers, phys = `physical  position (Mbp)`,
         gen = `genetic position (cM)`) %>%
  mutate(phys = phys * 10^6) %>%
  select(set, map, mkr, phys, gen) %>%
  mutate(vld = TRUE)
## Write to input file for MareyMap
write.table(
  df,
  file = "./ath/mm_input.txt",
  quote = TRUE,        # Enclose values in double quotes
  sep = " ",           # Separate values by white space
  row.names = FALSE    # Do not include row names in the output
)

## pseudogene data
pseudogenes <- read_tsv(here("pseudogene_df.tsv")) %>%
  filter(species == "Arabidopsis thaliana")
## we take the middle of the pseudogene as position of the pseudogene
df_psgs <- pseudogenes %>% 
  mutate(position_pseudogene = ((start + end) / 2)) %>%
  select(chrom, position_pseudogene, Type, pseudogene_id)

Run MareyMap

startMareyMapGUI()

We use the same strategy as Brazier and Glemin, 2022:

  • Two-degree polynomial loess regression

  • Span = 0.2

  • Negative recombination rates set to 0

Get resulting output file:

Bin per 500 kb windows and plot result

# Split recombination rate data in bins of 500 kb
df$bin <- cut_width(df$phys, width = 500000)

# Get average local recombination rate by taking the average of the loess estimate
# for the markers in that bin
mean_rec <- df %>% group_by(map, bin) %>%
  summarize(mean_GR_rate = mean(loess, na.rm = TRUE)) %>%
  mutate(map = paste0("Chr", map)) # same naming as pseudogene dataframe
## `summarise()` has grouped output by 'map'. You can override using the `.groups`
## argument.
# Split genome in 100 kb windows (30,427,671 is largest size of chromosome)
bins <- tibble(bin = cut_width(0:30427671, width = 500000)) %>% 
  unique()
map <- tibble(map = c("Chr1", "Chr2", "Chr3", "Chr4", "Chr5"))
bins <- expand_grid(bins, map, .name_repair = "minimal")

# Add recombination rate data to bins dataframe
bins_2 <- bins %>% left_join(mean_rec, by = c("bin", "map")) # Without changing NA to 0
bins <- bins %>% left_join(mean_rec, by = c("bin", "map")) %>%
  mutate(mean_GR_rate = ifelse(is.na(mean_GR_rate), 0, mean_GR_rate))

# Split pseudogene data in bins of 500 kb
df_psgs$bin <- cut_width(df_psgs$position_pseudogene, width = 500000)
# Count number of pseudogenes per bin
count_psgs <- df_psgs %>% group_by(chrom, bin, Type) %>%
  count(bin)
count_psgs_total <- df_psgs %>% group_by(chrom, bin) %>%
  count(bin, .drop = FALSE) %>%
  mutate(Type = "total")
count_psgs <- rbind(count_psgs, count_psgs_total)

# Add pseudogene data to bins dataframe
bins <- bins %>% left_join(count_psgs, by = c("bin", "map" = "chrom")) %>%
  rename(psgs = n) %>%
  mutate(psgs = ifelse(is.na(psgs), 0, psgs))
bins_2 <- bins_2 %>% left_join(count_psgs, by = c("bin", "map" = "chrom")) %>%
  rename(psgs = n) %>%
  mutate(psgs = ifelse(is.na(psgs), 0, psgs))

# If there is no estimate of local recombination, remove it from the analysis
bins_2 <- bins_2 %>%
  filter(!is.na(mean_GR_rate))

corr_plot <- ggplot(bins_2 %>%
                      filter(Type == "total"), aes(x = psgs, y = mean_GR_rate)) +

cor_table <- bins_wide %>%
  correlate(method = "pearson") %>%
  select(-c(bin, mean_GR_rate)) %>%
  filter(term == "mean_GR_rate")
## Non-numeric variables removed from input: `map`
## Correlation computed with
##  Method: 'pearson'
##  Missing treated using: 'pairwise.complete.obs'
calc_pval <- function(vec_a, vec_b, meth){
  cor.test(vec_a, vec_b, method = meth)$p.value
}

pval_table <- bins_wide %>%
  select(-c(bin, map)) %>%
  colpair_map(calc_pval, meth = "pearson") %>%
  select(-mean_GR_rate) %>%
  filter(term == "mean_GR_rate")

cor_table_spear <- bins_wide %>%
